# Char

## Interface
You can get ASCII control charaxters from property of this object.
<pre>
interface CharIF {
  etx : string ;
  eot : string ;
  bel : string ;
  bs : string ;
  tab : string ;
  lf : string ;
  vt : string ;
  cr : string ;
  esc : string ;
  del : string ;
}
